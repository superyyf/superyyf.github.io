二分查找的简单总结

August 3, 2021

二分查找

Tips

  1. 搜索空间可以选择左闭右开区间，或者两端都闭区间。
  2. 注意搜索空间的变化，while的终止条件要和搜索空间对应。
  3. 如果搜索左右边界，只要在input[mid]==target时，改变搜索空间即可。

代码实现

#include <iostream>
#include <vector>

using namespace std;

//无重复元素
//返回目标值的下标，没有目标值返回-1
int binarySearch_basic(const vector<int>& input, int target) {

	//搜索空间为两端都闭区间：[left, right]
	int left = 0;
	int right = input.size() - 1;
	
	//终止条件为left<=right,即当left>right时退出循环
	//终止条件要和搜索空间对应，[right, left]为空
	while (left <= right) {
		int mid = (left + right) / 2;
		if (input[mid] == target) return mid;
	
		//搜索空间变为[mid+1, right]
		else if (input[mid] < target) left = mid + 1;
	
		//搜索空间变为[left, mid-1]
		else if (input[mid] > target) right = mid - 1;
	}
	
	return -1;
}

//有重复元素，查找目标值的左边界
int binarySearch_left_bound(const vector<int>& input, int target) {

	//搜索空间为左闭右开区间：[left,right)
	int left = 0;
	int right = input.size();
	
	//终止条件为left < right，即当left == right时退出循环
	//[left,left)为空
	while (left < right) {
		int mid = (left + right) / 2;
	
		//由于寻找的是目标值的左边界，所有当mid处的值等于target时向左搜索
		//搜索空间为[left,mid)
		if (input[mid] == target) right = mid;
	
		//搜索空间变为[mid+1,right)
		else if (input[mid] < target) left = mid + 1;
	
		//搜索空间变为[left,mid)
		else if (input[mid] < target) right = mid;
	}
	
	//当退出循环时left==right，搜索空间为[left,left)
	//left下标就是目标值的左边界
	if (left == input.size())return -1;
	
	return input[left] == target ? left : -1;

}

//有重复元素，查找目标值的右边界
int binarySearch_right_bound(const vector<int>& input, int target) {

	//[left,right)
	int left = 0;
	int right = input.size();
	
	//[left,left)
	while (left < right) {
		int mid = (left + right) / 2;
	
		//[mid+1,right)
		if (input[mid] == target) left = mid + 1;
	
		//[mid+1,right)
		else if (input[mid] < target) left = mid + 1;
	
		//[left,mid)
		else if (input[mid] > target) right = mid;
	}
	
	if (left == input.size())return -1;
	return input[left - 1] == target ? left - 1 : -1;

}

int main() {
	vector<int> input = {1,2,2,2,2,3,4,5};
	int target = 2;
	cout << "basic: " << binarySearch_basic(input, target) << endl;
	cout << "left bound: " << binarySearch_left_bound(input, target) << endl;
	cout << "right bound: " << binarySearch_right_bound(input, target) << endl;
}
